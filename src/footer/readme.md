        <td>socialLinks: object[]</td>
        <td><code>[]</code></td>

            </ul>
        </td>
    </tr>

        <td>bottomLinks: object[]</td>

            <ul>
                <li><b>text</b>: Text for the link</li>
                <li><b>href</b>: URL of the page the link goes to</li>
            </ul>
        </td>
    </tr>
